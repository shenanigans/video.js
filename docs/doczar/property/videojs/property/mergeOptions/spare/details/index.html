spare .videojs.mergeOptions~details

Destructively merge options Objects, recursing through plain Objects and overwriting all other types. For example:

var defaultOptions = {
  foo: true,
  bar: {
    a: true,
    b: [ 1, 2, 3 ]
  }
};
var newOptions = {
  foo: false,
  bar: {
    b: [ 4 ]
  }
};
videojs.mergeOptions (defaultOptions, newOptions);
// defaultOptions.foo = false;
// defaultOptions.bar.a = true;
// defaultOptions.bar.b = [ 4 ];
